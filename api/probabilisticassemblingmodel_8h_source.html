220  void updateGeneUsage(const ClonesetView &cloneset) {
221  prob_t laplace = 0;
222  ClonesetView nonc = cloneset.noncoding();
223  if (_recomb == VJ_RECOMB) {
224  // Update V-J
225  _param_vec->familyFill(VJ_VAR_JOI_GEN, 0);
226  for (size_t i = 0; i < nonc.size(); ++i) {
227  uint n_vj = nonc[i].nVar() * nonc[i].nJoi();
228  for (seg_index_t v_i = 0; v_i < nonc[i].nVar(); ++v_i) {
229  for (seg_index_t j_i = 0; j_i < nonc[i].nJoi(); ++j_i) {
230  (*_param_vec)[_param_vec->event_index(VJ_VAR_JOI_GEN,
231  0,
232  nonc[i].getVar(v_i) - 1,
233  nonc[i].getJoi(j_i) - 1)] += 1. / n_vj;
234  }
235  }
236  }
237  _param_vec->normaliseEventClass(VJ_VAR_JOI_GEN);
238  } else if (_recomb == VDJ_RECOMB) {
239  // Update V
240  _param_vec->familyFill(VDJ_VAR_GEN, 0);
241  for (size_t i = 0; i < nonc.size(); ++i) {
242  for (seg_index_t v_i = 0; v_i < nonc[i].nVar(); ++v_i) {
243  (*_param_vec)[_param_vec->event_index(VDJ_VAR_GEN,
244  0,
245  nonc[i].getVar(v_i) - 1)] += 1. / nonc[i].nVar();
246  }
247  }
248  _param_vec->normaliseEventClass(VDJ_VAR_GEN);
249 
250  // Update J-D
251  _param_vec->familyFill(VDJ_JOI_DIV_GEN, 0);
252  for (size_t i = 0; i < nonc.size(); ++i) {
253  uint n_jd = nonc[i].nJoi() * nonc[i].nDiv();
254  for (seg_index_t j_i = 0; j_i < nonc[i].nJoi(); ++j_i) {
255  for (seg_index_t d_i = 0; d_i < nonc[i].nDiv(); ++d_i) {
256  (*_param_vec)[_param_vec->event_index(VDJ_JOI_DIV_GEN,
257  0,
258  nonc[i].getJoi(j_i) - 1,
259  nonc[i].getDiv(d_i) - 1)] += 1. / n_jd;
260  }
261  }
262  }
263  _param_vec->normaliseEventClass(VDJ_JOI_DIV_GEN);
264  }
265  }

342  void parseModelConfig(const string& folderpath) {

344 
345  ifstream ifs;
346  ifs.open(jsonpath);
347  _status = false;

349  ifs >> _config;

351  _recomb = VJ_RECOMB;

353  } else if (_config.get("recombination", "undefined").asString() == "VDJ") {
354  _recomb = VDJ_RECOMB;
355  _status = true;
356  } else {
357  _recomb = UNDEF_RECOMB;
358  _status = false;

360 
361  cout << "Probabilistic assembling model:\n\t" <<
362  _config.get("name", "Nameless model").asString() <<
363  "\n\t(" <<
364  _config.get("comment", "").asString() <<
365  ")\n\t" <<
366  "Path : " << folderpath <<
367  "\n\t" <<
368  "Specimen : " << _config.get("specimen", "hobbit").asString() <<
369  "\n\t" <<
370  _config.get("recombination", "undefined").asString() <<
371  "-recombination | " <<
372  ((_config.get("errors", 0).asDouble() != 0) ? "Sequence error model" : "No sequence error model") <<
373  endl << "\tFiles:"<< endl;
374 
375  _status = true;

377  std::cout << "[ERROR] Probabilistic assembling model error:" << endl << "\t config .json file not found at [" << jsonpath << "]" << std::endl;
